SELECT N.*, P.LANGUAGE_CD FROM (SELECT 
(SELECT PORTAL_LABEL FROM PSPRSMDEFNLANG WHERE PORTAL_OBJNAME = E.PORTAL_OBJNAME AND PORTAL_NAME = E.PORTAL_NAME AND LANGUAGE_CD = 'DUT')
|| ' >> ' || (SELECT PORTAL_LABEL FROM PSPRSMDEFNLANG WHERE PORTAL_OBJNAME = D.PORTAL_OBJNAME AND PORTAL_NAME = D.PORTAL_NAME AND LANGUAGE_CD = 'DUT')
|| ' >> ' || (SELECT PORTAL_LABEL FROM PSPRSMDEFNLANG WHERE PORTAL_OBJNAME = C.PORTAL_OBJNAME AND PORTAL_NAME = C.PORTAL_NAME AND LANGUAGE_CD = 'DUT')
|| ' >> ' || (SELECT PORTAL_LABEL FROM PSPRSMDEFNLANG WHERE PORTAL_OBJNAME = B.PORTAL_OBJNAME AND PORTAL_NAME = B.PORTAL_NAME AND LANGUAGE_CD = 'DUT')
|| ' >> ' || (SELECT PORTAL_LABEL FROM PSPRSMDEFNLANG WHERE PORTAL_OBJNAME = A.PORTAL_OBJNAME AND PORTAL_NAME = A.PORTAL_NAME AND LANGUAGE_CD = 'DUT')
FROM PSPRSMDEFN A LEFT JOIN PSPRSMDEFN B ON B.PORTAL_NAME = A.PORTAL_NAME
AND B.PORTAL_OBJNAME = A.PORTAL_PRNTOBJNAME LEFT JOIN PSPRSMDEFN C 
ON C.PORTAL_NAME = B.PORTAL_NAME AND C.PORTAL_OBJNAME = B.PORTAL_PRNTOBJNAME LEFT JOIN PSPRSMDEFN D 
ON D.PORTAL_NAME = C.PORTAL_NAME AND D.PORTAL_OBJNAME = C.PORTAL_PRNTOBJNAME LEFT JOIN PSPRSMDEFN E 
ON E.PORTAL_NAME = D.PORTAL_NAME AND E.PORTAL_OBJNAME = D.PORTAL_PRNTOBJNAME WHERE
A.PORTAL_OBJNAME = 'EOFSPC_QUERY_PGLT') N, PSPRSMDEFNLANG P;
